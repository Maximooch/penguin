   from prompt_toolkit.key_binding import KeyBindings
   from prompt_toolkit.keys import Keys


   def _create_prompt_session(self):
       """Create prompt session with @ file picker support"""
       kb = KeyBindings()


       @kb.add('@')
       def _(event):
           """Trigger file picker on @ key"""
           # Insert @ character first
           event.current_buffer.insert_text('@')


           # Check if we should auto-trigger picker
           text = event.current_buffer.text
           if text.strip().endswith('@'):
               # Schedule picker launch
               event.app.create_background_task(
                   self._launch_file_picker(event.current_buffer)
               )


       return PromptSession(
           multiline=True,
           key_bindings=kb,
           prompt_continuation='... '
       )

   import questionary
   from pathlib import Path
   from typing import List


   class FilePicker:
       def __init__(self, workspace_path: Path):
           self.workspace = workspace_path
           self.gitignore_patterns = self._load_gitignore()


       async def show_picker(self, start_dir: Optional[Path] = None) -> List[Path]:
           """Show interactive file picker"""
           current_dir = start_dir or self.workspace


           while True:
               # Get files in current directory
               choices = self._build_choices(current_dir)


               # Show picker
               selected = await questionary.checkbox(
                   f"Select files (in {current_dir.relative_to(self.workspace)}):",
                   choices=choices
               ).ask_async()


               if selected is None:  # User cancelled
                   return []


               # Handle directory navigation
               if '..' in selected:
                   current_dir = current_dir.parent
                   continue


               # Convert selections to paths
               return [current_dir / s for s in selected if s != '..']


       def _build_choices(self, directory: Path) -> List[dict]:
           """Build questionary choices from directory contents"""
           choices = []


           # Add parent directory option
           if directory != self.workspace:
               choices.append({
                   'name': '📁 .. (parent directory)',
                   'value': '..'
               })


           # Add subdirectories
           for item in sorted(directory.iterdir()):
               if item.is_dir() and not self._is_ignored(item):
                   choices.append({
                       'name': f'📁 {item.name}/',
                       'value': item.name
                   })


           # Add files
           for item in sorted(directory.iterdir()):
               if item.is_file() and not self._is_ignored(item):
                   icon = self._get_file_icon(item)
                   size = self._format_size(item.stat().st_size)
                   choices.append({
                       'name': f'{icon} {item.name} ({size})',
                       'value': item.name
                   })


           return choices

   async def _process_file_references(self, user_input: str) -> str:
       """Process @ file references in user input"""
       # Extract all @ references
       refs = re.findall(r'@([^\s]+)', user_input)


       loaded_files = []
       for ref in refs:
           if ref == '':  # Bare @ - launch picker
               files = await self.file_picker.show_picker()
               loaded_files.extend(files)
           elif '*' in ref:  # Glob pattern
               files = self._expand_glob(ref)
               loaded_files.extend(files)
           else:  # Direct reference
               file_path = self._resolve_path(ref)
               if file_path:
                   loaded_files.append(file_path)


       # Load files into context
       for file_path in loaded_files:
           await self._load_file_into_context(file_path)


       # Update user input with resolved paths
       updated_input = user_input
       for ref in refs:
           # Replace @ref with actual loaded files summary
           updated_input = updated_input.replace(
               f'@{ref}',
               f'[Loaded: {", ".join(f.name for f in loaded_files)}]'
           )


       return updated_input
